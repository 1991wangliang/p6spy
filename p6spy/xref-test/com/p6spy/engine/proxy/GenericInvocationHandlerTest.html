61      invocationHandler.addDelegate(new MethodNameMatcher("methodA"), delegate);
62      ExceptionHandling proxy = ProxyFactory.createProxy(targetObj, ExceptionHandling.class, invocationHandler);
63  
64      // unchecked exceptions will be passed through

68        fail("No exception thrown");
69      } catch(Exception e) {
70        assertEquals("Wrong exception thrown!",IllegalArgumentException.class, e.getClass());
71      }
72  
73      // declared checked exception will be passed through only if they are declared by the interface
74      try {
75        delegate.setExceptionToThrow(new IOException("blah"));
76        proxy.methodA();
77        fail("No exception thrown");
78      } catch(Exception e) {
79        assertEquals("Wrong exception thrown", UndeclaredThrowableException.class, e.getClass());

81      }
