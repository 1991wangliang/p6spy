48    @Before
49    public void activateOutage() throws SQLException {
50      // disable outage, as procedure creation takes already quite a while
51      P6OutageOptions.getActiveInstance().setOutageDetection("false");
52      Statement statement = connection.createStatement();
53      statement.execute(SLEEP_ALIAS);
54      P6OutageOptions.getActiveInstance().setOutageDetection("true");
55    }

57    @Test
58    public void testOutage() throws SQLException {
59      // exec fast query => no outage detected
60      callSleep(1);
61      Assert.assertFalse(super.getLastButOneLogEntry().contains("OUTAGE"));
62      Assert.assertFalse(super.getLastLogEntry().contains("OUTAGE"));
63      Assert.assertTrue(super.getLastLogEntry().contains("CALL SLEEP"));
64  
65      // exec slooooow query => outage detected
66      callSleep(2000);
67      Assert.assertTrue(super.getLastButOneLogEntry().contains("OUTAGE"));
68      Assert.assertTrue(super.getLastLogEntry().contains("CALL SLEEP"));
69    }
70  
71    private void callSleep(long miliseconds) throws SQLException {
72      final Statement statement = connection.createStatement();
73      statement.executeQuery("CALL SLEEP( " + miliseconds + ")");
