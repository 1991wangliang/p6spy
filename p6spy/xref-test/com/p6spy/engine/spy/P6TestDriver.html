89  	    if (driver.getClass().getName().equals(JDBC_DRIVER_CLASS_NAME_MYSQL)) {

92  	    }
93      }

95      /**
96       * Gets the wrapped {@link Driver} registered for the jdbc url test currently run for.
97       *
98       * @return the driver registered for the current run.
99       * @throws IOException 
100      * @throws SQLException
101      */
102     private Driver getWrappedDriver() throws IOException, SQLException {
103     	Properties props = loadProperties(p6TestProperties);
104     	String url = props.getProperty("url");
105     	Driver driver = DriverManager.getDriver(url);
106 
107 	    // make sure you have a p6 driver
108 	    if (! (driver instanceof P6SpyDriverCore)) {
109 	        fail("Expected to get back a p6spy driver, got back a " + driver);
110 	    }
111 	    return ((P6SpyDriverCore) driver).getPassthru();
112     }
