104 @RunWith(Parameterized.class)
105 public class P6TestBasics extends P6TestFramework {

107     public P6TestBasics(String db) throws SQLException, IOException {
108       super(db);
109     }

111     /* (non-Javadoc)
112      * @see com.p6spy.engine.spy.P6TestFramework#setUpFramework()
113      */
114     @Override
115     // just prevent inherited stuff
116     public void setUpFramework() {
117     }

119     @After
120     public void tearDown() throws SQLException {
121       if (this.connection != null) {
122         // close connection properly
123         this.connection.close();
124       }
125     }
126     
127     @Test
128     public void testNative() throws Exception {
129         connection = P6TestUtil.loadDrivers(null);
130         sqltests();
131     }
132     
133     @Test
134     public void testSpy() throws Exception {
135         List<String> driverNames = P6SpyOptions.getActiveInstance().getDriverNames();
136         if (null != driverNames && !driverNames.isEmpty()) {
137           connection = P6TestUtil.loadDrivers(driverNames.get(0));
138           sqltests();
139         } 
140     }
141 
142     protected void preparesql() throws SQLException {
143         final Statement statement = connection.createStatement();
144         drop(statement);
145         statement.execute("create table basic_test (col1 varchar(255), col2 integer)");
146         statement.close();
147     }
148 
149     protected void sqltests() throws SQLException {
150         preparesql();
151 
152         // insert test
153         String insert = "insert into basic_test values (\'bob\', 5)";
154         Statement statement = connection.createStatement();
155         statement.executeUpdate(insert);
156 
157         // update test
158         String update = "update basic_test set col1 = \'bill\' where col2 = 5";
159         statement.executeUpdate(update);
160 
161         // query test
162         String query = "select col1 from basic_test where col2 = 5";
163         ResultSet rs = statement.executeQuery(query);
164         rs.next();
165         assertEquals(rs.getString(1), "bill");
166 
167         // prepared test
168         PreparedStatement ps = connection.prepareStatement("insert into basic_test values (?, ?)");
169         ps.setString(1,"joe");
170         ps.setInt(2,6);
171         ps.executeUpdate();
172         ps.setString(1,"andy");
173         ps.setInt(2,7);
174         ps.execute();
175 
176         ps = connection.prepareStatement("update basic_test set col1 = ? where col2 = ?");
177         ps.setString(1,"charles");
178         ps.setInt(2,6);
179         ps.executeUpdate();
180         ps.setString(1,"bobby");
181         ps.setInt(2,7);
182         ps.execute();
183 
184         ps = connection.prepareStatement("select col1 from basic_test where col1 = ? and col2 = ?");
185         ps.setString(1,"charles");
186         ps.setInt(2,6);
187         rs = ps.executeQuery();
188         rs.next();
189         assertEquals("charles", rs.getString(1));
190         
191         ps.close();
192     }
193 
194     protected void drop(Statement statement) {
195       try {
196         if (statement == null) {
197           return;
198         }
199         statement.execute("drop table basic_test");
200       } catch (Exception e) {
201         // we don't really care about cleanup failing
202       }
203     }
