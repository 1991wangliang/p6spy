204   public void resetLoadedDrivers() throws SQLException, IOException {
205     List<Driver> dereg = new ArrayList<Driver>();
206     for (Enumeration<Driver> e = DriverManager.getDrivers(); e.hasMoreElements();) {
207       Driver driver = e.nextElement();
208 
209       // get rid of all driver stuff to start with the clean table
210       dereg.add(driver);
211     }
212 
213     // if you found any drivers let's dereg them now
214     for (Driver driver : dereg) {
215       DriverManager.deregisterDriver(driver);
216     }
217 
218     // make sure to reinit with the db file related to current run
219     Map tp = getDefaultPropertyFile();
220     reloadProperty(tp);
221 
222     // make sure to reinit for each Driver run as we run parametrized builds
223     // and need to have fresh stuff for every specific driver
224     P6SpyDriverCore.initialized = false;
225     P6SpyDriver.initMethod();
226   }

262     protected Properties loadProperties(String filename) throws IOException {
263         if (filename == null) {
264             throw new IllegalArgumentException("No properties file specified.");
265         }
266 
267         Properties props = new Properties();
268 
269         InputStream inputStream = this.getClass().getResourceAsStream(filename);
270         if ( inputStream == null ) {
271             inputStream = new FileInputStream(filename);
272         }
273         try {
274             props.load(inputStream);
275         } finally {
276             try {
277                 inputStream.close();
278             } catch(Exception e) {
279                 // so earlier exception is not shadowed.
280             }
281         }
282         return props;
283     }

285     protected void writeProperty(String filename, Map<String,String> props) {
286         try {
287             File reload = new File(filename);
288             reload.delete();
289 
290             PrintWriter out = new PrintWriter(new BufferedWriter(new FileWriter(reload)));
291 
292             for(Map.Entry<String, String>entry: props.entrySet()) {
293                 out.println(entry.getKey()+"="+entry.getValue());
294             }
295 
296             out.close();
297         } catch (Exception e) {
298             fail(e.getMessage());
299         }
300     }

380     protected void reloadProperty(Map props) {
381         try {
382             writeProperty(PROPERTY_FILE, props);
383             P6SpyProperties properties = new P6SpyProperties();
384             properties.setSpyProperties(PROPERTY_FILE);
385             OptionReloader.reload();
386         } catch (Exception e) {
387             fail(e.getMessage());
388         }

391     protected void assertIsLastQuery(String query) {
392         boolean isTrue = P6LogQuery.getLastEntry().contains(query);
393         if (!isTrue) {
394             System.err.println(query+" was not the last query, this was: "+P6LogQuery.getLastEntry());
395         }
396         assertTrue(isTrue);
397     }

399     protected void assertIsNotLastQuery(String query) {
400         boolean isFalse = !P6LogQuery.getLastEntry().contains(query);
401         if (!isFalse) {
402             System.err.println(query+" was the last query and should not have been");
403         }
