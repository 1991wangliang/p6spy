62      Assert.assertNull(opts.getDateformat());
63      Assert.assertEquals(FileLogger.class.getName(), opts.getAppender());
64      Assert.assertEquals(P6SpyFactory.class.getName(), opts.getModulelist());
65      Assert.assertEquals(1, opts.getModuleFactories().size());
66      Assert.assertTrue(opts.getModuleFactories().iterator().next() instanceof P6SpyFactory);
67      Assert.assertEquals(1, opts.getModuleNames().size());
68      Assert.assertTrue(opts.getModuleNames().contains(P6SpyFactory.class.getName()));

70      Assert.assertNull(opts.getDriverNames());

74      Assert.assertFalse(opts.getReloadProperties());

84    @Test
85    public void testP6LogOptionDefaults() {
86      // make sure to have relevant module loaded
87      {
88        System.setProperty(SystemProperties.P6SPY_PREFIX + P6SpyOptions.MODULELIST,
89            P6LogFactory.class.getName());
90        P6SpyOptions.getActiveInstance().reload();
91      }
92  
93      final P6LogLoadableOptions opts = P6LogOptions.getActiveInstance();
94      Assert.assertNotNull(opts);
95  
96      Assert.assertNull(opts.getSQLExpression());
97      Assert.assertEquals(0L, opts.getExecutionThreshold());
98      Assert.assertNull(opts.getIncludecategories());
99      Assert.assertEquals("info,debug,result,resultset,batch", opts.getExcludecategories());
100     Assert.assertNull(opts.getIncludeCategoriesSet());
101     Assert.assertTrue(opts.getExcludeCategoriesSet().containsAll(
102         Arrays.asList(new String[] { "info", "debug", "result", "resultset", "batch" })));
103     Assert.assertFalse(opts.getFilter());
104     Assert.assertNull(opts.getIncludeTables());
105     Assert.assertNull(opts.getExcludeTables());
