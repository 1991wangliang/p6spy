91  import static org.junit.Assert.assertEquals;
92  import static org.junit.Assert.fail;

94  import java.io.IOException;
95  import java.sql.CallableStatement;
96  import java.sql.SQLException;

98  import org.junit.Ignore;
99  import org.junit.Test;
100 import org.junit.runner.RunWith;
101 import org.junit.runners.Parameterized;
102 
103 @RunWith(Parameterized.class)
104 public class P6TestCallableStatement extends P6TestPreparedStatement {
105 
106 	public P6TestCallableStatement(String db) throws SQLException, IOException {
107     super(db);
108   }
109 
110   @Ignore
111     @Test
112     public void testCallable () throws Exception {
113        int bigParam = 1024;
114        int valuesLen;
115        int setterMax = 32;
116        StringBuffer testproc = new StringBuffer(bigParam);
117 
118        // set
119        testproc.append("CALL TEST.METHOD (");
120        for (int i = 0; i < bigParam; i++) {
121          testproc.append("?");
122        }
123        testproc.append(")");
124 
125        CallableStatement call = connection.prepareCall(testproc.toString());
126 
127        for (int x=1; x<=setterMax;x++){
128         String tmpstring = ("String" +x);
129         call.setString(x, tmpstring);
130         }
131        
132        setterMax++;
133        
134        try {
135        call.registerOutParameter(setterMax,java.sql.Types.INTEGER);
136        // values should be grown after this call since
137        // setterMax is greater than array length
138        valuesLen = ((P6CallableStatement)call).getValuesLength();
139        assertEquals(setterMax+P6CallableStatement.P6_GROW_MAX, valuesLen);
140 
141        //  try various registerOutParameter methods
142        call.registerOutParameter(1,java.sql.Types.INTEGER);
143 
144        setterMax+=P6CallableStatement.P6_GROW_MAX;
145        call.registerOutParameter(setterMax,java.sql.Types.FLOAT,3);
146        // values should be grown after this call since
147        // setterMax is greater than array length
148        valuesLen = ((P6CallableStatement)call).getValuesLength();
149        assertEquals(setterMax+P6CallableStatement.P6_GROW_MAX, valuesLen);
150 
151        call.registerOutParameter(setterMax+3,java.sql.Types.INTEGER);
152        call.registerOutParameter(bigParam,java.sql.Types.INTEGER);
153        } catch (Exception e)  {
154          fail(e.getMessage()+" Failed Registering Out Parameter: " + getStackTrace(e));
155        }
156 
157        // last register out with param of bigParam will cause
158        // values to be grown based on bigParam.  Test follows...
159        valuesLen = ((P6CallableStatement)call).getValuesLength();
160        assertEquals(bigParam+P6CallableStatement.P6_GROW_MAX, valuesLen);
161        call.close();
162     }
