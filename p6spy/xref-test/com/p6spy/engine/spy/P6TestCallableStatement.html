111 @RunWith(Parameterized.class)
112 public class P6TestCallableStatement extends P6TestPreparedStatement {
113 
114   private static final Collection<Object[]> DBS_IN_TEST = Arrays.asList(new Object[][] { { "H2" } });
115   
116   /**
117    * Always returns {@link DBS_IN_TEST} as we don't
118    * need to rerun for each DB here.
119    * The thing is that not all the DBs support stored procedures. Morever syntax might differ. 
120    * We want just to prove that callable statements are correctly prxied.
121    * So let's test just with H2 (default DB).
122    * 
123    * @return {@link DBS_IN_TEST}
124    */
125   @Parameters
126   public static Collection<Object[]> dbs() {
127     return DBS_IN_TEST;
128   }
129 
130   
131 	public P6TestCallableStatement(String db) throws SQLException, IOException {
132     super(db);
133   }
134 	  
135     @Test
136     public void testCallable() throws SQLException {
137       
138       // tests inspired by: http://opensourcejavaphp.net/java/h2/org/h2/test/jdbc/TestCallableStatement.java.html
139       Statement stat = connection.createStatement();
140       {
141         stat.execute("CREATE TABLE TEST(ID INT, NAME VARCHAR)");
142         CallableStatement call = connection.prepareCall("INSERT INTO TEST VALUES(?, ?)");
143         call.setInt(1, 1);
144         call.setString(2, "Hello");
145         call.execute();
146       
147         assertTrue(((P6TestLogger) P6LogQuery.getLogger()).getLastEntry().indexOf("INSERT INTO TEST VALUES") != -1);
148       }
149      
150       {
151         CallableStatement call = connection.prepareCall("SELECT * FROM TEST", ResultSet.TYPE_FORWARD_ONLY, ResultSet.CONCUR_READ_ONLY);
152         ResultSet rs = call.executeQuery();
153         rs.next();
154         assertEquals(1, rs.getInt(1));
155         assertEquals("Hello", rs.getString(2));
156         assertFalse(rs.next());
157         
158         assertTrue(((P6TestLogger) P6LogQuery.getLogger()).getLastEntry().indexOf("SELECT * FROM TEST") != -1);
159       }
160       
161       {
162         CallableStatement call = connection.prepareCall("SELECT * FROM TEST", ResultSet.TYPE_FORWARD_ONLY, ResultSet.CONCUR_READ_ONLY, ResultSet.HOLD_CURSORS_OVER_COMMIT);
163         ResultSet rs = call.executeQuery();
164         rs.next();
165         assertEquals(1, rs.getInt(1));
166         assertEquals("Hello", rs.getString(2));
167         assertFalse(rs.next());
168 
169         assertTrue(((P6TestLogger) P6LogQuery.getLogger()).getLastEntry().indexOf("SELECT * FROM TEST") != -1);
170       }
171       
172   }
