10  /**
11   * SQL batch file logger Private: (?) No Appender that writes a trace of JDBC activity into an SQL
12   * batch file that can be later "replayed" using a generic SQL client.
13   * <p>
14   * modify the spy.properties file to make p6spy use this appender by adding line
15   * appender=com.p6spy.engine.logging.appender.BatchFileLogger and commenting out the existing
16   * appender #appender=com.p6spy.engine.logging.appender.FileLogger - specify an alternative log file
17   * name in spy.properties so that diagnostic entries to spy.log made by p6spy at startup do not mess
18   * up your batch.
19   * </p>
20   * <p>Here's how it works for me: logfile = spy.sql The appender writes the text of SQL
21   * statements as well as commit and rollback commands to the specified file, each on a new line. For
22   * prepared statements the effective text resulting from substitution of parameter signs with their
23   * values is written. Other event categories are logged as SQL comments ("-- " followed by the
24   * category name). Exceptions and text are silently discarded. All SQL statements except the last
25   * one will have a delimiter character ';' appended to them. This batch format works well with
26   * WinSQL. The format is hardcoded, so if you want to make it configurable you'll have to patch my
27   * patch or write your own :)
28   * */

39      @Override
40      public void logSQL(int connectionId, String now, long elapsed, String category, String prepared, String sql) {
41          if (endOfStatement) {
42              qlog.println(BATCH_SEPARATOR);
43          }
44          if (category.equals("statement")) {
45              String actual = null == sql || 0 == sql.length() ? prepared : sql;
46              qlog.print(actual);
47              endOfStatement = true;
48          } else if ("commit".equalsIgnoreCase(category) || "rollback".equalsIgnoreCase(category)) {
49              qlog.print(category.toUpperCase());
50              endOfStatement = true;
51          } else {
52              qlog.println("-- " + category);
53              endOfStatement = false;
54          }
55          qlog.flush();
56      }

59      public void logText(String text) {
60      }

62      @Override
63      public void setLogfile(String fileName) {
64          try {
65              boolean append = P6SpyOptions.getAppend();
66              endOfStatement = append && 0L < new File(fileName).length();
67              qlog = new PrintStream(new FileOutputStream(fileName, append));
68          } catch (IOException e) {
69              e.printStackTrace(System.err);
70          }
71      }
