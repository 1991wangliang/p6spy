112     public SubclassTask() {
113         fileSets = new ArrayList<FileSet>();
114     }
115 
116     @Override
117     public void execute() throws BuildException {
118         // use the fileset defined in the parent (abstract) class
119         loader = createClassLoader();
120 
121         for (FileSet fileset : fileSets) {
122             DirectoryScanner ds = fileset.getDirectoryScanner(getProject());
123             String[] files = ds.getIncludedFiles();
124 
125             if ( files != null) {
126                 for (String fileName: files) {
127                     if (fileName.endsWith(".class")) {
128                         fixAndLoad(fileName);
129                     } else if (fileName.endsWith(".jar")) {
130                         expandJar(ds.getBasedir(), fileName);
131                     } else {
132                         log("File '" + fileName + "' is neither a .class nor a .jar file, skipping.", Project.MSG_WARN);
133                     }
134                 }
135             }
136         }
137     }
138     public void fixAndLoad(String fileName) {
139         // strip off the ".class" at the end
140         String cleanString = fileName.substring(0, fileName.indexOf(".class"));
141         StringBuffer className = new StringBuffer();
142 
143         // change all of the delims to dots... don't use the
144         // replace methods on String to preserve some backwards compat
145         int start = 0;
146         int pos = 0;
147         while ((pos = cleanString.indexOf(Subclasser.DELIMITER, start)) > 0) {
148             className.append(cleanString.substring(start, pos));
149             className.append(".");
150             start = pos + 1;
151         }
152 
153         // catch the tail
154         if (start < cleanString.length()) {
155             className.append(cleanString.substring(start));
156         }
157 
158         cleanString = className.toString();
159         // try not to further subclass P6 Stuff
160         // this is a little weak, but whatever
161         if (cleanString.indexOf(".P6") == -1) {
162             loadClass(cleanString);
163         }
164     }
165 
166     public void loadClass(String className) {
167         try {
168             Class k = loader.loadClass(className);
169             int mods = k.getModifiers();
170 
171             // make sure it's a DataSource and something
172             // instantiatable (whew)
173             if (DataSource.class.isAssignableFrom(k) && ((mods & Modifier.ABSTRACT) == 0) && ((mods & Modifier.INTERFACE) == 0)
174                 && ((mods & Modifier.FINAL) == 0)) {
175 
176                 log("Found class " + className, Project.MSG_WARN);
177 
178                 Subclasser sub = new Subclasser(k);
179 
180                 if (outputPackage != null) {
181                     sub.setOutputPackage(outputPackage);
182                 }
183 
184                 if (outputDir != null) {
185                     sub.setOutputDir(outputDir);
186                 }
187 
188                 sub.createSubClass();
189             }
190         } catch (Exception e) {
191             throw new BuildException(e);
192         }
193     }
194 
195     public void expandJar(File parentDir, String jarName) {
196         try {
197             File jarFS = new File(parentDir, jarName);
198             JarFile jar = new JarFile(jarFS);
199 
200             for (Enumeration e = jar.entries(); e.hasMoreElements();) {
201                 JarEntry entry = (JarEntry) e.nextElement();
202 
203                 if (!entry.isDirectory() && entry.getName().endsWith(".class")) {
204                     fixAndLoad(entry.getName());
205                 }
206 
207             }
208         } catch (IOException e) {
209             throw new BuildException(e);
210         }
211 
212     }

218         return loader;
