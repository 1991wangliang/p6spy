97      /** Creates a new instance of P6SpyProperties */
98      public P6SpyProperties() {
99          // we only reload the properties information if it has changed
100         File propertiesFile = new File(propertiesPath);
101         if(propertiesFile.exists()) {
102             long lastModified = propertiesFile.lastModified();
103             if(lastModified != propertiesLastModified) {
104                 propertiesLastModified = lastModified;
105                 properties = P6Util.loadProperties(SPY_PROPERTIES_FILE);
106             } else {
107                 properties = null;
108             }
109         }
110     }

127     /* gets the full path, including filename, of the property file */
128     protected static String findPropertiesPath() {
129         String propertiesPath = P6Util.classPathFile(SPY_PROPERTIES_FILE);
130         if (propertiesPath != null) {
131             File propertiesFile = new File(propertiesPath);

139     // save the file down
140     public static void saveProperties() {
141         Properties props = new Properties();
142         for (P6Options opts : OptionReloader.options) {
143             Method[] allMethods = opts.getClass().getDeclaredMethods();
144             for (Method m : allMethods) {
145                 if (!m.getName().startsWith("get")) {
146                     continue;
147                 }
148                 Class[] params = m.getParameterTypes();
149                 if (params.length != 0) {
150                     continue;
151                 }
152                 String propertyName = m.getName().substring(3).toLowerCase();
153                     Object rv;
154                     try {
155                         rv = m.invoke(opts);
156                         if (rv != null) {
157                             props.setProperty(propertyName, rv.toString());
158                             P6LogQuery.info("added property '" + propertyName + "' with value of '" + rv + "'");
159                         }
160                     } catch (IllegalAccessException e ) {
161                       throw new IllegalStateException(e);
162                     } catch(IllegalArgumentException e ) {
163                       throw new IllegalStateException(e);
164                     } catch( InvocationTargetException e) {
165                         throw new IllegalStateException(e);
166                     }
167             }
168         }
169 
170         // now your props object should have everything you want,
171         // so open up the file and write out the data
172         // NB I don't let you change the name of your properties... maybe
173         // in the future
174 
175         //->JAW removed this line of code -- File propertiesFile = new File(propertiesPath);
176         //since its only use was in a constructor that is only available in JDK1.4.
177         //Also then changed the offending constructor and placed close into finally
178         //block where it belongs.  Also fixed wording of error message.
179 
180         FileOutputStream out = null;
181         try {
182             out = new FileOutputStream(propertiesPath, false);
183             props.store(out, "P6Spy configuration");
184             out.flush();
185             P6LogQuery.info("successfully saved properties to file " + propertiesPath);
186             //propertiesLastModified = propertiesFile.lastModified();
187         } catch (Exception e) {
188             P6LogQuery.error("Could not save to property file " + propertiesPath + " because of error " + e);
189         } finally {
190             try {
191                 if (out != null) {
192                     out.close();
193                 }
194             } catch (IOException ioe) {
195                 P6LogQuery.error("Could not close property file " + propertiesPath + " because of error " + ioe);
196             }
197         }
198         //->JAW
199     }

214     public void setClassValues(Class klass) {
215         // only invoke this if the property file has been changed
216         if (properties == null) {
217             return;
218         }
219 
220         // first load the properties from the property file
221         List<String> allMethods = P6Util.findAllMethods(klass);
222         for(String m : allMethods) {
223             // lowercase and strip the end
224             String methodName = m.substring(3);
225             String value = (String)properties.get(methodName.toLowerCase());
226             P6Util.dynamicSet(klass, "set"+methodName, value == null ? null : value.trim());
227         }
228 
229         // next, check the environment and see if we should override any properties
230         Collection<String> list = P6Util.findAllMethods(klass);
231 
232         P6LogQuery.info("Using properties file: "+propertiesPath);
233 
234         for (String opt : list) {
235             String value = System.getProperty("p6" + opt);
236 
237             if (value != null) {
238                 P6LogQuery.info("Found value in environment: "+opt+", setting to value: "+value);
239                 P6Util.dynamicSet(klass, opt,value);
240             } else {
241                 P6LogQuery.info("No value in environment for: "+opt+", using: "+P6Util.dynamicGet(klass,opt));
242             }
243         }
244     }

246     public List<String> getOrderedList(String prefix) {
247         List<String> orderedList = new ArrayList<String>();
248         List<KeyValue> list = P6Util.loadProperties(SPY_PROPERTIES_FILE, prefix);
249         for (KeyValue keyValue : list) {
250             String value = keyValue.getValue();
251             orderedList.add(value);
252         }
253         return orderedList;
254     }
