63    /**
64     * 
65     * @param spyPropertiesJMX
66     *          manually (via JMX) set properties to be kept across auto-reloads.
67     */
68    private synchronized static void initMe() {
69      try {
70        cleanUp();
71  
72        instance = new P6ModuleManager();
73        P6LogQuery.initialize();
74        
75        // get rid of old cached stuff
76        GenericInvocationHandler.clearCache();
77        
78      } catch (IOException e) {
79        handleInitEx(e);
80      } catch (MBeanRegistrationException e) {
81        handleInitEx(e);
82      } catch (InstanceNotFoundException e) {
83        handleInitEx(e);
84      } catch (MalformedObjectNameException e) {
85        handleInitEx(e);
86      } catch (InstanceAlreadyExistsException e) {
87        handleInitEx(e);
88      } catch (NotCompliantMBeanException e) {
89        handleInitEx(e);
90      }
91    }

93    private static void cleanUp() throws MBeanRegistrationException, InstanceNotFoundException,
94        MalformedObjectNameException {
95      if (instance == null) {
96        return;
97      }
98  
99      for (P6OptionsSource optionsSource : instance.optionsSources) {
100       optionsSource.preDestroy(instance);
101     }
102 
103     // unregister mbeans (to prevent naming conflicts)
104     if (instance.mBeansRegistry != null) {
105       instance.mBeansRegistry.unregisterAllMBeans();
106     }
107   }

109   /**
110    * Used on the class load only (only once!)
111    * 
112    * @throws IOException
113    * @throws NotCompliantMBeanException
114    * @throws MBeanRegistrationException
115    * @throws InstanceAlreadyExistsException
116    * @throws MalformedObjectNameException
117    */
118   private P6ModuleManager() throws IOException, InstanceAlreadyExistsException,
119                            MBeanRegistrationException, NotCompliantMBeanException,
120                            MalformedObjectNameException {
121     debug(this.getClass().getName() + " re/initiating modules started");
122 
123     // make sure the proper listener registration happens
124     registerOptionChangedListener(new P6LogQuery());
125     
126     // hard coded - core module init - as it holds initial config
127     final P6SpyLoadableOptions spyOptions = (P6SpyLoadableOptions) registerModule(new P6SpyFactory());
128     loadDriversExplicitly(spyOptions);
129     
130     // configured modules init
131     for (P6Factory factory : spyOptions.getModuleFactories()) {
132     	registerModule(factory);
133     }
134 
135     optionsRepository.initCompleted();
136     
137     for (P6OptionsSource optionsSource : optionsSources) {
138       optionsSource.postInit(this);
139     }
140 
141     debug(this.getClass().getName() + " re/initiating modules done");
142   }

144   protected synchronized P6LoadableOptions registerModule(P6Factory factory) throws InstanceAlreadyExistsException, 
145   	MBeanRegistrationException, NotCompliantMBeanException, MalformedObjectNameException {
146 	  // re-register is not supported - skip silently
147 	  for (P6Factory registeredFactory : factories) {
148 		  if (registeredFactory.getClass().equals(factory.getClass())) {
149 			  return null;
150 		  }
151 	  }
152 		  
153 	  final P6LoadableOptions options = factory.getOptions(optionsRepository);
154       loadOptions(options);
155       
156       allOptions.put(options.getClass(), options);
157 	  factories.add(factory);
158       mBeansRegistry.registerMBean(options);
159       
160       debug("Registered factory: " + factory.getClass().getName() + " with options: " + options.getClass().getName());
161       
162       return options;
163   }

165   /**
166    * Returns loaded options. These are loaded in the right order:
167    * <ul>
168    * <li>default values</li>
169    * <li>based on the order defined in the {@link #optionsSources}</li>
170    * </ul>
171    * 
172    * @param options
173    * @return
174    */
175 	private void loadOptions(final P6LoadableOptions options) {
176 		// make sure to load defaults first
177 		options.load(options.getDefaults());
178 
179 		// load the rest in the right order then
180 		for (P6OptionsSource optionsSource : optionsSources) {
181 			Map<String, String> toLoad = optionsSource.getOptions();
182 			if (null != toLoad) {
183 				options.load(toLoad);
184 			}
185 		}
186 
187 		// register to all the props then
188 		allOptions.put(options.getClass(), options);
189 	}

199 	private void loadDriversExplicitly(P6SpyLoadableOptions spyOptions) {
200 		final Collection<String> driverNames = spyOptions.getDriverNames();
201 		if (null != driverNames) {
202 			for (String driverName : driverNames) {
203 				try {
204 					// you really only need to load the driver if it is not a
205 					// type 4 driver!
206 					P6Util.forName(driverName).newInstance();
207 				} catch (Exception e) {
208 					String err = "Error registering driver names: "
209 							+ driverNames + " \nCaused By: " + e.toString();
210 					P6LogQuery.error(err);
211 					throw new P6DriverNotFoundError(err);
212 				}
213 			}
214 		}
215 	}

217   private void debug(String msg) {
218     // not initialized yet => nowhere to log yet
219     if (instance == null || factories.isEmpty()) {
220       return;
221     }

223     P6LogQuery.debug(msg);
