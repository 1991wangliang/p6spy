20  package com.p6spy.engine.spy;
21  
22  import java.lang.management.ManagementFactory;
23  import java.util.ArrayList;
24  import java.util.Collection;
25  
26  import javax.management.InstanceAlreadyExistsException;
27  import javax.management.InstanceNotFoundException;
28  import javax.management.MBeanRegistrationException;
29  import javax.management.MBeanServer;
30  import javax.management.MalformedObjectNameException;
31  import javax.management.NotCompliantMBeanException;
32  import javax.management.ObjectName;
33  import javax.management.StandardMBean;
34  
35  public class P6MBeansRegistry {
36  
37    private final Collection<ObjectName> mBeans = new ArrayList<ObjectName>();
38  
39    public static final String PACKAGE_NAME = "com.p6spy";
40    
41    public void registerMBeans(Collection<P6LoadableOptions> allOptions) throws MBeanRegistrationException, InstanceNotFoundException, MalformedObjectNameException, NotCompliantMBeanException {
42      boolean jmx = true; 
43      String jmxPrefix = "";
44      
45      for (P6LoadableOptions options : allOptions) {
46        if (options instanceof P6SpyOptions) {
47          jmx = ((P6SpyOptions) options).getJmx();
48          jmxPrefix = ((P6SpyOptions) options).getJmxPrefix();
49          break;
50        }
51      }
52         
53      if (!jmx) {
54        return;
55      }
56      
57      // unreg possible conflicting ones first
58      unregisterAllMBeans(jmxPrefix);
59      
60      // reg all
61      for (P6LoadableOptions options : allOptions) {
62        try {
63          registerMBean(options, jmxPrefix);
64        } catch (InstanceAlreadyExistsException e) {
65          // sounds like someone registered beans already (before we had a chance to do so)
66          // so let's just make things consistent and re-register again
67          registerMBeans(allOptions);
68        }
69      }
70    }
71  
72    protected void registerMBean(P6LoadableOptions mBean, String jmxPrefix) throws InstanceAlreadyExistsException,
73        MBeanRegistrationException, NotCompliantMBeanException, MalformedObjectNameException {
74  
75      checkMBean(mBean);
76  
77      final ObjectName mBeanObjectName = getObjectName(mBean, jmxPrefix);
78      ManagementFactory.getPlatformMBeanServer().registerMBean(mBean, mBeanObjectName);
79      mBeans.add(mBeanObjectName);
80    }
81  
82    public void unregisterAllMBeans(String jmxPrefix) throws MBeanRegistrationException, MalformedObjectNameException {
83  
84      // those we have reference to 
85      final MBeanServer mbs = ManagementFactory.getPlatformMBeanServer();
86      for (ObjectName mBeanObjectName : mBeans) {
87        try {
88          mbs.unregisterMBean(mBeanObjectName);
89        } catch (InstanceNotFoundException e) {
90          // this just means someone unregistered our beans already
91          // but we're OK with that and it can't cause failure
92        }
93      }
94      mBeans.clear();
95  
96      // to prevent naming conflicts: let's unreg also possible leftovers (with the same prefix)
97      for (ObjectName objectName : mbs.queryNames(new ObjectName(getPackageName(jmxPrefix) + ":name=com.p6spy.*"), null)) {
98        try {
99          mbs.unregisterMBean(objectName);
100       } catch (InstanceNotFoundException e) {
101         // this just means someone unregistered the bean earlier than us
102         // (quite unprobable, but parallel unreg could happen)
103         // but we're OK with that and it can't cause failure
104       }
105       
106     }
107   }
108 
109   private void checkMBean(P6LoadableOptions mBean) {
110     if (null == mBean) {
111       throw new IllegalArgumentException("mBean is null!");
112     }
113 
114     if (!(mBean instanceof StandardMBean)) {
115       throw new IllegalArgumentException(
116           "mBean has to be instance of the StandardMBean class! But is not: " + mBean);
117     }
118   }
119 
120   protected ObjectName getObjectName(P6LoadableOptions mBean, String jmxPrefix) throws MalformedObjectNameException {
121     return new ObjectName(getPackageName(jmxPrefix) + ":name=" + mBean.getClass().getName());
122   }
123   
124   protected static String getPackageName(String jmxPrefix) {
125     return PACKAGE_NAME +  (null == jmxPrefix || jmxPrefix.isEmpty() ? "" : "." + jmxPrefix);
126   }
