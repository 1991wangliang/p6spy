84      @Override
85      public void load(Map<String, String> options) {
86        loadLog4jConfig(P6Util.getProperties(options));
87  
88        if (options.get(LOG_MESSAGE_FORMAT) != null) {
89          setLogMessageFormat(options.get(LOG_MESSAGE_FORMAT));
90        }
91        if (options.get(LOGFILE) != null) {
92          setLogfile(options.get(LOGFILE));
93        }
94        if (options.get(APPEND) != null) {
95          setAppend(options.get(APPEND));
96        }
97        if (options.get(DATEFORMAT) != null) {
98          setDateformat(options.get(DATEFORMAT));
99        }
100       if (options.get(APPENDER) != null) {
101         setAppender(options.get(APPENDER));
102       }
103       if (options.get(MODULELIST) != null) {
104         setModulelist(options.get(MODULELIST));
105       }
106       if (options.get(DRIVERLIST) != null) {
107         setDriverlist(options.get(DRIVERLIST));
108       }
109       if (options.get(STACKTRACE) != null) {
110         setStackTrace(options.get(STACKTRACE));
111       }
112       if (options.get(STACKTRACECLASS) != null) {
113         setStackTraceClass(options.get(STACKTRACECLASS));
114       }
115       //      setUsePrefix(options.get(USE_PREFIX));
116       if (options.get(AUTOFLUSH) != null) {
117         setAutoflush(options.get(AUTOFLUSH));
118       }
119       if (options.get(RELOADPROPERTIES) != null) {
120         setReloadProperties(options.get(RELOADPROPERTIES));
121       }
122       if (options.get(RELOADPROPERTIESINTERVAL) != null) {
123         setReloadPropertiesInterval(options.get(RELOADPROPERTIESINTERVAL));
124       }
125       if (options.get(JNDICONTEXTFACTORY) != null) {
126         setJNDIContextFactory(options.get(JNDICONTEXTFACTORY));
127       }
128       if (options.get(JNDICONTEXTPROVIDERURL) != null) {
129         setJNDIContextProviderURL(options.get(JNDICONTEXTPROVIDERURL));
130       }
131       if (options.get(JNDICONTEXTCUSTOM) != null) {
132         setJNDIContextCustom(options.get(JNDICONTEXTCUSTOM));
133       }
134       if (options.get(REALDATASOURCE) != null) {
135         setRealDataSource(options.get(REALDATASOURCE));
136       }
137       if (options.get(REALDATASOURCECLASS) != null) {
138         setRealDataSourceClass(options.get(REALDATASOURCECLASS));
139       }
140       if (options.get(REALDATASOURCEPROPERTIES) != null) {
141         setRealDataSourceProperties(options.get(REALDATASOURCEPROPERTIES));
142       }
143       if (options.get(DATABASE_DIALECT_DATE_FORMAT) != null) {
144         setDatabaseDialectDateFormat(options.get(DATABASE_DIALECT_DATE_FORMAT));
145       }
146     }

148     /**
149      * Loads log4j specific configuration.
150      * <br/><br/>
151      * Please note: The existing configuration is not cleared nor reset. It's rather iterative approach here
152      * once you require different behavior, provide your own log4j configuration file (holding these properties) 
153      * and make sure to load/reload it properly.
154      * 
155      * @param properties the properties to load the configuration values from.
156      */
157     private void loadLog4jConfig(Properties properties) {
158       PropertyConfigurator.configure(properties);
159     }
