18  import java.io.File;
19  import java.io.FileOutputStream;
20  import java.io.IOException;
21  import java.io.PrintStream;

23  import com.p6spy.engine.spy.P6SpyOptions;

25  /**
26   * SQL batch file logger Private: (?) No Appender that writes a trace of JDBC activity into an SQL
27   * batch file that can be later "replayed" using a generic SQL client.
28   * <p>
29   * modify the spy.properties file to make p6spy use this appender by adding line
30   * appender=com.p6spy.engine.logging.appender.BatchFileLogger and commenting out the existing
31   * appender #appender=com.p6spy.engine.logging.appender.FileLogger - specify an alternative log file
32   * name in spy.properties so that diagnostic entries to spy.log made by p6spy at startup do not mess
33   * up your batch.
34   * </p>
35   * <p>Here's how it works for me: logfile = spy.sql The appender writes the text of SQL
36   * statements as well as commit and rollback commands to the specified file, each on a new line. For
37   * prepared statements the effective text resulting from substitution of parameter signs with their
38   * values is written. Other event categories are logged as SQL comments ("-- " followed by the
39   * category name). Exceptions and text are silently discarded. All SQL statements except the last
40   * one will have a delimiter character ';' appended to them. This batch format works well with
41   * WinSQL. The format is hardcoded, so if you want to make it configurable you'll have to patch my
42   * patch or write your own :)
43   * */
44  public class BatchFileLogger extends FileLogger {
45  
46      public BatchFileLogger() {
47          //this("spy.sql");
48      }
49  
50      @Override
51      public void logException(Exception e) {
52      }
53  
54      @Override
55      public void logSQL(int connectionId, String now, long elapsed, String category, String prepared, String sql) {
56          if (endOfStatement) {
57              qlog.println(BATCH_SEPARATOR);
58          }
59          if (category.equals("statement")) {
60              String actual = null == sql || 0 == sql.length() ? prepared : sql;
61              qlog.print(actual);
62              endOfStatement = true;
63          } else if ("commit".equalsIgnoreCase(category) || "rollback".equalsIgnoreCase(category)) {
64              qlog.print(category.toUpperCase());
65              endOfStatement = true;
66          } else {
67              qlog.println("-- " + category);
68              endOfStatement = false;
69          }
70          qlog.flush();
71      }
72  
73      @Override
74      public void logText(String text) {
75      }
76  
77      @Override
78      public void setLogfile(String fileName) {
79          try {
80              boolean append = P6SpyOptions.getActiveInstance().getAppend();
81              endOfStatement = append && 0L < new File(fileName).length();
82              qlog = new PrintStream(new FileOutputStream(fileName, append));
83          } catch (IOException e) {
84              e.printStackTrace(System.err);
85          }
86      }
87  
88      public static final char BATCH_SEPARATOR = ';';
89  
90      private boolean endOfStatement;
91  }
