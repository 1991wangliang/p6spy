28  /**
29   * Appender which uses Log4J configured via spy.properties.
30   * <p>
31   *   Note: This appender is deprecated in P6Spy 2.0.  For similar functionality,
32   *   use the Slf4JLogger instead.
33   * </p>
34   *
35   */
36  // adapted from Rafael Alvarez's LoggingStream class
37  @Deprecated
38  public class Log4jLogger extends FormattedLogger implements P6Logger {
39      
40      protected Level level = Level.INFO;
41      private static Logger log;
42      
43      // By configuring log4j by this method, we control the p6spy logger behavior
44      // using the same configuration file as p6spy, and any other attempt
45      // to configure log4j will add to this configuration.
46      public Log4jLogger() {
47        // [Peter Butkovic] functionality moved to P6LogOptions.load()
48  //        P6SpyProperties properties = new P6SpyProperties();
49  //        P6LogLoadableOptions options = P6ModuleManager.getInstance().getOptions(P6LogLoadableOptions.class);
50  //        PropertyConfigurator.configure(options.getLog4JProperties());
51          log = Logger.getLogger("p6spy");
52          log.setAdditivity(false);
53      }
54      
55      public void logException(Exception e) {
56          StringWriter sw = new StringWriter();
57          PrintWriter pw  = new PrintWriter(sw);
58          e.printStackTrace(pw);
59          logText(sw.toString());
60      }
61      
62      public void logText(String text) {
63          log.log(this.level, text);
64      }
65      
66      public Level getLevel() {
67          return this.level;
68      }
69      
70      public void setLevel(Level inVar) {
71          this.level = inVar;
72      }   
73  }
