22  import com.p6spy.engine.logging.Category;
23  
24  // log4j, stdout, and file logger all use the same format
25  // so we descend from this class.
26  public abstract class FormattedLogger {
27    private MessageFormattingStrategy strategy;
28  
29    protected FormattedLogger() {
30      strategy = new SingleLineFormat();
31    }
32  
33    public void logSQL(int connectionId, String now, long elapsed, String category, String prepared, String sql) {
34      logText(strategy.formatMessage(connectionId, now, elapsed, category, prepared, sql));
35    }
36  
37    public void logSQL(int connectionId, String now, long elapsed, Category category, String prepared, String sql) {
38  	    logText(strategy.formatMessage(connectionId, now, elapsed, category.toString(), prepared, sql));
39    }
40    
41    public abstract void logText(String text);
42  
43    /**
44     * Sets the strategy implementation to use for formatting log message.  If not set, this will default to {@link SingleLineFormat}
45     */
46    public void setStrategy(final MessageFormattingStrategy strategy) {
47      this.strategy = strategy;
48    }
49  }
