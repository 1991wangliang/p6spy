26  public class MultiLineFormat implements MessageFormattingStrategy {
27  
28    /**
29     * Formats a log message for the logging module
30     *
31     * @param connectionId the id of the connection
32     * @param now          the current ime expressing in milliseconds
33     * @param elapsed      the time in milliseconds that the operation took to complete
34     * @param category     the category of the operation
35     * @param prepared     the SQL statement with all bind variables replaced with actual values
36     * @param sql          the sql statement executed
37     * @return the formatted log message
38     */
39    @Override
40    public String formatMessage(final int connectionId, final String now, final long elapsed, final String category, final String prepared, final String sql) {
41      return "#" + now + " | took " + elapsed + "ms | " + category + " | connection " + connectionId + "|" + prepared + "\n" + sql +";";
42    }
43  }
