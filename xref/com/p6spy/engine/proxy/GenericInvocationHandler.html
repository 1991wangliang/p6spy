16  package com.p6spy.engine.proxy;
17  
18  import java.lang.reflect.InvocationHandler;
19  import java.lang.reflect.Method;
20  import java.util.HashMap;
21  import java.util.Map;
22  
23  /**
24   * Base class for invocation handlers.  This class is designed to be a generic implementation
25   * of the {@link InvocationHandler} interface which delegates the invocation of {@link Delegate} objects
26   * based on pattern matching against the method.
27   *
28   * @param <T> The class of the object which will be proxied
29   */
30  public class GenericInvocationHandler<T> implements InvocationHandler {
31    private final Map<MethodMatcher, Delegate> delegateMap;
32  
33    private final T underlying;
34  
35    /**
36     * Creates a new invocation handler for the given object.
37     *
38     * @param underlying  The object being proxied
39     */
40    public GenericInvocationHandler(T underlying) {
41      this.underlying = underlying;
42      this.delegateMap = new HashMap<MethodMatcher, Delegate>();
43    }
44  
45    /**
46     * Adds a delegate which will be used when a method is invoked that meets the following criteria:
47     * <p>
48     *   <code>methodMatcher.match(method) == true</code>
49     * </p>
50     * Note: Adding a second delegate object with the same method matcher (according to {@link MethodMatcher#equals(Object)}
51     * and {@link com.p6spy.engine.proxy.MethodMatcher#hashCode()} will replace the previous delegate!
52     *
53     * @param methodMatcher The method matcher
54     * @param delegate The delegate object
55     */
56    public void addDelegate(MethodMatcher methodMatcher, Delegate delegate) {
57      delegateMap.put(methodMatcher, delegate);
58    }
59  
60    Delegate getDelegate(MethodMatcher methodMatcher) {
61      // This is package scope to allow test code to invoke the method
62      return delegateMap.get(methodMatcher);
63    }
64  
65    public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
66      for (MethodMatcher methodMatcher : delegateMap.keySet()) {
67        if (methodMatcher.matches(method)) {
68          return delegateMap.get(methodMatcher).invoke(underlying, method, args);
69        }
70      }
71      
72      return method.invoke(underlying, args);
73    }
74  
75    protected T getUnderlying() {
76      return underlying;
77    }
78  
79  }
