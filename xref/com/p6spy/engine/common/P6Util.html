318     protected static void removeDots(Properties props) {
319         Map<String, String> hash     = new HashMap<String, String>();
320         boolean done     = false;
321 
322         for(Enumeration<Object> keys = props.keys(); keys.hasMoreElements();) {
323             String key = (String) keys.nextElement();
324             if (key.indexOf('.') != -1) {
325                 int len    = key.length();
326                 int newLen = 0;
327                 char[] car = new char[len];
328                 for (int i = 0; i < len; i++) {
329                     char c = key.charAt(i);
330                     if (c != '.') {
331                         car[newLen++] = c;
332                     }
333                 }
334 
335                 String out = new String(car, 0, newLen);
336                 hash.put(out, props.getProperty(key));
337             }
338         }
339 
340         if (done) {
341             props.putAll(hash);
342         }
343     }

367     // this is our own version, which we need to do to ensure the order is kept
368     // in the property file
369     public static List<KeyValue> loadProperties(String file, String prefix) {
370         List<KeyValue> props = new ArrayList<KeyValue>();
371         FileReader in = null;
372         BufferedReader reader = null;
373 
374         try {
375             String path = classPathFile(file);
376 
377             if (path == null) {
378                 P6LogQuery.error("Can't find " + file + ". "+getCheckedPath());
379             } else {
380                 in = new FileReader(path);
381                 // read the file
382                 reader = new BufferedReader(in);
383                 String line;
384                 while ((line = reader.readLine()) != null) {
385                     if ((line.trim()).startsWith(prefix)) {
386                         StringTokenizer st = new StringTokenizer(line, "=");
387                         try {
388                             String name = st.nextToken();
389                             String value = st.nextToken();
390                             KeyValue kv = new KeyValue(name.trim(), value.trim());
391                             props.add(kv);
392                         } catch (NoSuchElementException e) {
393                             // ignore; means that you have
394                             // something like:
395                             // realdriver2=
396                         }
397                     }
398                 }
399             }
400         } catch (FileNotFoundException e1) {
401             P6LogQuery.error("File not found " + file + " " + e1);
402         } catch (IOException e2) {
403             P6LogQuery.error("IO Error reading file " + file + " " + e2);
404         } finally {
405             try {
406                 if (reader != null) {
407                     reader.close();
408                 }
409                 if (in != null) {
410                     in.close();
411                 }
412             } catch (IOException e) {
413             }
414         }
415         return props;
416     }

418     /**
419      * Here we attempt to find the file in the current dir and the classpath
420      * If we can't find it then we return null
421      */
422     public static String classPathFile(String file) {
423         File fp             = null;
424         String path         = null;
425         String separator    = System.getProperty("path.separator");
426         String slash        = System.getProperty("file.separator");
427         String classpath    = "." + separator + System.getProperty("java.class.path");
428         String local        = System.getProperty("p6.home");
429 
430         // first try to load options via the classloader
431         try {
432             // If p6.home is specified, just look there
433             if (local != null) {
434                 fp = new File(local, file);
435             } else {
436                 // try to get the classloader via the current thread
437                 fp = classLoadPropertyFile(Thread.currentThread().getContextClassLoader().getResource(file));
438 
439                 if (fp == null) {
440                     // next try the current class
441                     fp = classLoadPropertyFile(P6Util.class.getClassLoader().getResource(file));
442                 }
443 
444                 if (fp == null) {
445                     // now the ClassLoader system resource
446                     classLoadPropertyFile(ClassLoader.getSystemResource(file));
447                 }
448             }
449 
450             if (fp.exists()) {
451                 return fp.getCanonicalPath();
452             }
453         } catch (Exception exc) {
454         }
455 
456         // if that failed, see what we can do on our own
457         StringTokenizer tok = new StringTokenizer(classpath, separator);
458 
459         do {
460             String dir = tok.nextToken();
461             path = dir.equals(".") ? file : dir + slash + file;
462             fp = new File(path);
463         } while (!fp.exists() && tok.hasMoreTokens());
464 
465         return fp.exists() ? path : null;
466     }

501     /**
502      * A utility for using the current class loader (rather than the
503      * system class loader) when instantiating a new class.
504      * <p>
505      * The idea is that the thread's current loader might have an
506      * obscure notion of what your class path is (e.g. an app server) that
507      * will not be captured properly by the system class loader.
508      * <p>
509      * taken from http://sourceforge.net/forum/message.php?msg_id=1720229
510      *
511      * @param name class name to load
512      * @return the newly loaded class
513      */
514     public static Class forName(String name) throws ClassNotFoundException {
515         ClassLoader ctxLoader = null;
516         try {
517             ctxLoader = Thread.currentThread().getContextClassLoader();
518             return Class.forName(name, true, ctxLoader);
519 
520         } catch(ClassNotFoundException ex) {
521             // try to fall through and use the default
522             // Class.forName
523             //if(ctxLoader == null) { throw ex; }
524         } catch(SecurityException ex) {
525         }
526         return Class.forName(name);
527     }
