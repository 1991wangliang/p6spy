16  package com.p6spy.engine.logging;
17  
18  import com.p6spy.engine.common.ConnectionInformation;
19  import com.p6spy.engine.common.StatementInformation;
20  import com.p6spy.engine.proxy.GenericInvocationHandler;
21  import com.p6spy.engine.proxy.MethodNameMatcher;
22  
23  import java.sql.Statement;
24  
25  
26  /**
27   * Invocation handler for {@link Statement}
28   */
29  class P6LogStatementInvocationHandler extends GenericInvocationHandler<Statement> {
30  
31    public P6LogStatementInvocationHandler(Statement underlying, final ConnectionInformation connectionInformation) {
32      super(underlying);
33      StatementInformation statementInformation = new StatementInformation(connectionInformation);
34  
35      P6LogStatementExecuteDelegate executeDelegate = new P6LogStatementExecuteDelegate(statementInformation);
36      P6LogStatementExecuteBatchDelegate executeBatchDelegate = new P6LogStatementExecuteBatchDelegate(statementInformation);
37      P6LogStatementAddBatchDelegate addBatchDelegate = new P6LogStatementAddBatchDelegate(statementInformation);
38  
39  
40      addDelegate(
41          new MethodNameMatcher("executeBatch"),
42          executeBatchDelegate
43      );
44      addDelegate(
45          new MethodNameMatcher("addBatch"),
46          addBatchDelegate
47      );
48      addDelegate(
49          new MethodNameMatcher("execute"),
50          executeDelegate
51      );
52      addDelegate(
53          new MethodNameMatcher("executeQuery"),
54          executeDelegate
55      );
56      addDelegate(
57          new MethodNameMatcher("executeUpdate"),
58          executeDelegate
59      );
60  
61    }
62  
63  }
