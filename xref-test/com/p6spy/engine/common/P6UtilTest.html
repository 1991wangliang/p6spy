34    @Test 
35    public void testJoinNullSafe() {
36      Assert.assertEquals("", P6Util.joinNullSafe(null, null));
37      Assert.assertEquals("", P6Util.joinNullSafe(null, ""));
38      Assert.assertEquals("", P6Util.joinNullSafe(null, ","));
39      Assert.assertEquals("", P6Util.joinNullSafe(Collections.<String>emptyList(), null));
40      Assert.assertEquals("", P6Util.joinNullSafe(Collections.<String>emptyList(), ","));
41      Assert.assertEquals("foo", P6Util.joinNullSafe(Arrays.asList("foo"), ","));
42      Assert.assertEquals("foobar", P6Util.joinNullSafe(Arrays.asList("foo", "bar"), null));
43      Assert.assertEquals("foo,bar", P6Util.joinNullSafe(Arrays.asList("foo", "bar"), ","));
44      Assert.assertEquals("foo|bar|aaa", P6Util.joinNullSafe(Arrays.asList("foo", "bar", "aaa"), "|"));
45    }

60    }
