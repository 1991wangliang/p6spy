39  public class P6WrapperIsWrapperDelegateTest extends BaseTestCase {
40  
41    @Test
42    public void testCastableFromProxy() throws SQLException {
43      Connection con = new TestConnectionImpl();
44      Connection proxy = ProxyFactory.createProxy(con, new GenericInvocationHandler<Connection>(con));
45  
46      // if the proxy implements the interface then true should be returned.
47      assertTrue(proxy.isWrapperFor(Connection.class));
48      assertTrue(proxy.isWrapperFor(TestConnection.class));
49      assertTrue(proxy.isWrapperFor(Wrapper.class));
50    }
51  
52    @Test
53    public void testCastableFromUnderlying() throws SQLException {
54      Connection con = new TestConnectionImpl();
55      Connection proxy = ProxyFactory.createProxy(con, new GenericInvocationHandler<Connection>(con));
56  
57      // if the underlying object extends the class (or matches the class) then true should be returned.
58      assertTrue(proxy.isWrapperFor(TestConnectionImpl.class));
59      assertTrue(proxy.isWrapperFor(AbstractTestConnection.class));
60    }
61  
62    @Test
63    public void testProxyOfWrappedConnection() throws SQLException {
64      // this will be the actual connection
65      Connection con = new TestConnectionImpl();
66  
67      // use a wrapper from DBCP to create a proxy of a proxy
68      // Note: DBCP implements with JDBC 4.0 API so the Wrapper interface
69      // is implemented here.
70      DelegatingConnection underlying = new DelegatingConnection(con);
71  
72      Connection proxy = ProxyFactory.createProxy(underlying, new GenericInvocationHandler<Connection>(underlying));
73  
74      // TestConnection is an interface of the wrapped underlying object.
75      assertTrue(proxy.isWrapperFor(TestConnection.class));
76  
77      // ResultSet is not implemented at all - false should be returned
78      assertFalse(proxy.isWrapperFor(ResultSet.class));
79  
80    }
