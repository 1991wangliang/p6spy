67      public void testSpy() throws Exception {
68          Collection<String> driverNames = P6SpyOptions.getActiveInstance().getDriverNames();
69          if (null != driverNames && !driverNames.isEmpty()) {
70            connection = P6TestUtil.loadDrivers(driverNames.iterator().next());
71            sqltests();
72          }

75      protected void preparesql() throws SQLException {
76          final Statement statement = connection.createStatement();
77          drop(statement);
78          statement.execute("create table basic_test (col1 varchar(255), col2 integer)");
79          statement.close();
80      }
81  
82      protected void sqltests() throws SQLException {
83          preparesql();
84  
85          // insert test
86          String insert = "insert into basic_test values (\'bob\', 5)";
87          Statement statement = connection.createStatement();
88          statement.executeUpdate(insert);
89  
90          // update test
91          String update = "update basic_test set col1 = \'bill\' where col2 = 5";
92          statement.executeUpdate(update);
93  
94          // query test
95          String query = "select col1 from basic_test where col2 = 5";
96          ResultSet rs = statement.executeQuery(query);
97          rs.next();
98          assertEquals(rs.getString(1), "bill");
99  
100         // prepared test
101         PreparedStatement ps = connection.prepareStatement("insert into basic_test values (?, ?)");
102         ps.setString(1,"joe");
103         ps.setInt(2,6);
104         ps.executeUpdate();
105         ps.setString(1,"andy");
106         ps.setInt(2,7);
107         ps.execute();
108 
109         ps = connection.prepareStatement("update basic_test set col1 = ? where col2 = ?");
110         ps.setString(1,"charles");
111         ps.setInt(2,6);
112         ps.executeUpdate();
113         ps.setString(1,"bobby");
114         ps.setInt(2,7);
115         ps.execute();
116 
117         ps = connection.prepareStatement("select col1 from basic_test where col1 = ? and col2 = ?");
118         ps.setString(1,"charles");
119         ps.setInt(2,6);
120         rs = ps.executeQuery();
121         rs.next();
122         assertEquals("charles", rs.getString(1));
123         
124         ps.close();
125     }
