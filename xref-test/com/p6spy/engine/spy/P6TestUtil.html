129     protected static Properties loadProperties(String filename) throws IOException {
130         if (filename == null) {
131             throw new IllegalArgumentException("No properties file specified.");
132         }
133 
134         Properties props = new Properties();
135 
136         InputStream inputStream = P6TestUtil.class.getResourceAsStream(filename);
137         if ( inputStream == null ) {
138             inputStream = new FileInputStream(filename);
139         }
140         try {
141             props.load(inputStream);
142         } finally {
143             try {
144                 inputStream.close();
145             } catch(Exception e) {
146                 // so earlier exception is not shadowed.
147             }
148         }
149         return props;
150     }

152     protected static void writeProperty(String filename, Map props) throws IOException {
153         File reload = new File(filename);
154         reload.delete();
155 
156         PrintWriter out = new PrintWriter(new BufferedWriter(new FileWriter(reload)));
157 
158         Iterator i = props.keySet().iterator();
159         while (i.hasNext()) {
160             String key = (String)i.next();
161             String value = (String)props.get(key);
162             out.println(key+"="+value);
163         }
164 
165         out.close();
166     }

228     protected static void assertIsLastQuery(String query) {
229         boolean isTrue = P6LogQuery.getLastEntry().indexOf(query) != -1;
230         if (!isTrue) {
231           LOG.error(query+" was not the last query, this was: "+P6LogQuery.getLastEntry());

250     protected static void unloadDrivers() throws Exception {
251         Properties props = loadProperties("P6Test.properties");
252         String drivername = props.getProperty("p6driver");
253         String user = props.getProperty("user");
254         String password = props.getProperty("password");
255         String url = props.getProperty("url");
256 
257         try {
258             Driver driver;
259             if ( url != null && url.length() > 0) {
260                 while ((driver = DriverManager.getDriver(url)) != null) {
261                     LOG.info("Deregistering driver: "+driver.getClass().getName());
262                     DriverManager.deregisterDriver(driver);
263                 }
264             }
265         } catch (Exception e) {
266         }
267     }

269     public static Connection loadDrivers(String drivernameProperty) throws SQLException, IOException, ClassNotFoundException, IllegalAccessException, InstantiationException {
270         Properties props = loadProperties(P6TestFramework.P6_TEST_PROPERTIES);
271         String drivername = props.getProperty(drivernameProperty);
272         String user = props.getProperty("user");
273         String password = props.getProperty("password");
274         String url = props.getProperty("url");
275 
276         LOG.info("UTIL REGISTERING DRIVER PROPERTY == "+drivernameProperty+" TO REGISTER DRIVER == "+drivername);
277         Class<Driver> driverClass = P6Util.forName(drivername);
278         DriverManager.setLogWriter(new PrintWriter(System.out, true));
279         // if a type 4 JDBC driver is unregistered, you must register it before using it!
280         DriverManager.registerDriver(driverClass.newInstance());
281         Driver driver = DriverManager.getDriver(url);
282         LOG.info("UTIL USING DRIVER == "+driver.getClass().getName()+" FOR URL "+url);
283         Connection connection = DriverManager.getConnection(url, user, password);
284         printAllDrivers();
285         return connection;
286     }
